#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define pb   push_back
void solve ()
{
  ll n;
  cin >> n;
  ll a[n];
  ll f=0;
  map<ll, ll > m;
  for (int i=0;i<n; i++)
  {
      cin >> a[i];
  }
  ll x=0;
  for (int i=n-1; i>=0; i--)
  {
      if (m[a[i]] == 0)
      {
          m[a[i]]++;
      }
      else {
            f++;
         x = i;
         break;
      }
 
  }
  if (f==0) cout << x << endl;
  else cout << x+1 << endl;
 
}
int main()
{
 ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    cin>> t;
    while (t--)
    {
         solve ();
    }
 
 
 
}
