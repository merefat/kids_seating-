#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define pb   push_back
void solve ()
{
   ll n;
   cin >> n;
   ll a[n];
   ll ans = 1;
   for (int i=0; i<n; i++)
   {
       cin >> a[i];
       ans*=a[i];
       
   }
   ans+=n-1;
   cout << ans *2022 << endl;
 
}

{

 int t;
 cin>> t;

 {
